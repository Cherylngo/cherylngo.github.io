freq=dict(df_pres['State'].value_counts())
fig,ax=plt.subplots()
state=list(freq.keys())
count=list(freq.values())
ax.bar(x = state[:8], height = count[:8], color='black')
fig.set_size_inches(8,5)
fig.suptitle('States With The Highest Number of Presidential Births')
ax.set_xlabel('State')
ax.set_ylabel('Number of Presidents Born')
plt.show()
